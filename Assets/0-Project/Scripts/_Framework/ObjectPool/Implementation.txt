Spawnlanacak Objeye PoolObject scripti ekle sonra böyle çağır.
SetActive false yapınca otomatikmen poola gider.


using UnityEngine;
using OWS.ObjectPooling;

public class SampleObjectPool : MonoBehaviour
{
	public static ObjectPool<PoolObject> objectPool;
	[SerializeField] private GameObject objectPrefab;

	private void Awake()
	{
		// create pool instance and send in prefab reference
		objectPool = new ObjectPool<PoolObject>(objectPrefab);
	}

	private void SomeFunction()
	{
		// get object from pool
		PoolObject newPoolObject objectPool.Pull();
	}
}
